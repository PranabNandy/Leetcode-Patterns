9. pow() function in O(logn) time

```c++
int power(int x, int y) {
int ans=1;
while(y>0){
       if(y%2){
           ans=ans*x;    // y==1
           y=y-1;
      }
      else{
        x=x*x;
        y=y/2;
      }
}
return ans;
```
